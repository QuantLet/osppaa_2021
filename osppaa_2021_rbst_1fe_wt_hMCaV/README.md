f_norm <- "lead(diff(prices)) ~

                                        # Model estimation
m1way_wT_hMCaV <- plm(formula = f_norm,
          data    = as.data.frame(dta),
          model   = 'within',
          effect = 'individual',
          index = c('coin'))


summary(m1way_wT_hMCaV,
        vcov = function(x) vcovSCC(x))
